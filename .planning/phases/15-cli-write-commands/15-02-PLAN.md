     ```typescript
     async function parseJsonInput(opts: Record<string, unknown>): Promise<Record<string, unknown>> {
       let raw: string;
       if (opts.data) {
         raw = opts.data as string;
       } else if (!process.stdin.isTTY) {
         const chunks: Buffer[] = [];
         for await (const chunk of process.stdin) { chunks.push(chunk as Buffer); }
         raw = Buffer.concat(chunks).toString('utf-8');
       } else {
         outputError({ error: 'JSON input required. Use --data flag or pipe JSON to stdin', code: 'VALIDATION_ERROR' }, ExitCode.VALIDATION);
       }
       try {
         return JSON.parse(raw) as Record<string, unknown>;
       } catch (e) {
         outputError({ error: 'Invalid JSON: ' + (e instanceof Error ? e.message : String(e)), code: 'VALIDATION_ERROR' }, ExitCode.VALIDATION);
       }
     }
     ```
